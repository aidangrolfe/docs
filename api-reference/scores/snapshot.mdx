---

Get snapshots for each action in the latest score events for a fixture.

## Path Parameters

<ParamField path="fixtureId" type="int64" required>
  The ID of the fixture
</ParamField>

## Query Parameters

<ParamField query="asOf" type="int64">
  Optional Unix timestamp (ms) for the latest time the historical snapshots are returned for. If omitted, returns the live snapshot.
</ParamField>

## Response

Returns an array of Scores objects containing score event data.

<ResponseField name="fixtureId" type="int32" required>
  Fixture identifier
</ResponseField>

<ResponseField name="gameState" type="string" required>
  Current state of the game
</ResponseField>

<ResponseField name="startTime" type="int64" required>
  Fixture start time timestamp
</ResponseField>

<ResponseField name="isTeam" type="boolean" required>
  Indicates if this is a team event
</ResponseField>

<ResponseField name="fixtureGroupId" type="int32" required>
  Fixture group identifier
</ResponseField>

<ResponseField name="competitionId" type="int32" required>
  Competition identifier
</ResponseField>

<ResponseField name="countryId" type="int32" required>
  Country identifier
</ResponseField>

<ResponseField name="sportId" type="int32" required>
  Sport identifier
</ResponseField>

<ResponseField name="participant1IsHome" type="boolean" required>
  Indicates if participant 1 is the home team
</ResponseField>

<ResponseField name="participant2Id" type="int32" required>
  Second participant identifier
</ResponseField>

<ResponseField name="participant1Id" type="int32" required>
  First participant identifier
</ResponseField>

<ResponseField name="action" type="string" required>
  Action type for this score event
</ResponseField>

<ResponseField name="id" type="int32" required>
  Event identifier
</ResponseField>

<ResponseField name="ts" type="int64" required>
  Timestamp of the score event
</ResponseField>

<ResponseField name="connectionId" type="int64" required>
  Connection identifier
</ResponseField>

<ResponseField name="seq" type="int32" required>
  Sequence number of the score event
</ResponseField>

<ResponseField name="stats" type="object">
  Map of statistical keys to their values
</ResponseField>
